class Solution:
    def threeSum(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        result = []
        for i, a in enumerate(nums):
            if a < 0:
                if -a in nums:
                    if 0 in nums:
                        if [a, 0, -a] not in result:
                            result.append([a, 0, -a])
                for j, b in enumerate(nums):
                    if b < 0 and i != j:
                        c = a + b
                        if -c in nums:
                            if [a, b, -c] not in result and [b, a, -c] not in result:
                                result.append([a, b, -c])
                    if b > 0:
                        c = -a - b
                        if c in nums[j+1:] and c > 0:
                            if [a, b, c] not in result and [a, c, b] not in result:
                                result.append([a, b, c])
            if a == 0:
                b = c = a
                if b in nums[i+1: ]:
                    n = nums[i+1: ].index(b) + (i+1)
                    if c in nums[n + 1: ]:
                        if [a, b, c] not in result:
                            result.append([a, b, c])
        return result
