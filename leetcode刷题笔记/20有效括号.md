class Solution:
    def isValid(self, s):
        """
        :type s: str
        :rtype: bool
        """
        rule = {'}': '{', ')': '(', ']': '['}
        
        if s == '':
            return True
        else:
            stack = []
            for i in range(len(s)):
                if s[i] in rule.values():
                    stack.append(s[i])
                elif s[i] in rule.keys():
                    if stack == []:
                        stack.append(s[i])
                    else:
                        if rule[s[i]] == stack[-1]:
                            stack.pop()
                        else:
                            stack.append(s[i])
            return not stack
